# 3. DATA WRANGLING (CRIAÇÃO DO PAINEL POR CATEGORIA)
# ==============================================================================
df_painel_cat <- df_items %>%
  inner_join(df_orders, by = "order_id") %>%
  inner_join(df_products, by = "product_id") %>%
  filter(!is.na(product_category_name)) %>%
  mutate(ano_mes = floor_date(order_purchase_timestamp, "month")) %>%
  group_by(product_category_name, product_id, ano_mes) %>%
  summarise(
    quantidade = n(),
    preco_medio = mean(price, na.rm = TRUE),
    frete_medio = mean(freight_value, na.rm = TRUE),
    .groups = "drop"
  ) %>%
  mutate(
    ln_preco = log(preco_medio),
    ln_frete = log(frete_medio + 1)
  ) %>%
  arrange(product_category_name, product_id, ano_mes) %>%
  group_by(product_id) %>%
  mutate(ln_preco_lag1 = lag(ln_preco, 1)) %>%
  filter(!is.na(ln_preco_lag1)) %>%
  filter(n() > 1) %>%
  ungroup()

# Isolar as Top 10 Categorias por volume
top_10_categorias <- df_painel_cat %>%
  count(product_category_name, sort = TRUE) %>%
  slice_head(n = 10) %>%
  pull(product_category_name)

df_top_cat <- df_painel_cat %>%
  filter(product_category_name %in% top_10_categorias)

# 4. MODELAGEM (ELASTICIDADE COM MÉTRICAS ESTATÍSTICAS COMPLETAS)
# ==============================================================================
cat("\n[1/3] Calculando Elasticidades e Métricas Completas de Avaliação...\n")
## 
## [1/3] Calculando Elasticidades e Métricas Completas de Avaliação...
tabela_elasticidade <- df_top_cat %>%
  split(.$product_category_name) %>%
  map_df(function(dados_categoria) {
    tryCatch({
      # Roda o modelo PPML
      modelo <- fepois(
        quantidade ~ ln_preco_lag1 + ln_frete | product_id + ano_mes,
        data = dados_categoria,
        cluster = ~product_id,
        warn = FALSE, notes = FALSE
      )
      
      # 1. Coeficientes, Erro Padrão e P-valor
      elasticidade <- coef(modelo)["ln_preco_lag1"]
      erro_padrao  <- se(modelo)["ln_preco_lag1"]
      p_valor      <- pvalue(modelo)["ln_preco_lag1"]
      
      # 2. Intervalo de Confiança (Aproximação 95%: Coef +/- 1.96 * SE)
      ci_inferior <- elasticidade - (1.96 * erro_padrao)
      ci_superior <- elasticidade + (1.96 * erro_padrao)
      
      # 3. Métricas de Ajuste e Tamanho da Amostra
      metricas  <- fitstat(modelo, c("pr2", "rmse", "aic", "bic"))
      n_obs     <- nobs(modelo)
      
      tibble(
        Categoria = unique(dados_categoria$product_category_name),
        N_Obs = n_obs,
        Elasticidade = round(elasticidade, 3),
        SE = round(erro_padrao, 3),
        CI_95_Inferior = round(ci_inferior, 3),
        CI_95_Superior = round(ci_superior, 3),
        P_Valor = round(p_valor, 4),
        Status = ifelse(p_valor < 0.05, "Significativo", "Não Significativo"),
        Pseudo_R2 = round(metricas$pr2, 4),
        RMSE = round(metricas$rmse, 2),
        AIC = round(metricas$aic, 1),
        BIC = round(metricas$bic, 1)
      )
    }, error = function(e) { NULL })
  })

# 5. SIMULADOR DE CENÁRIOS DE PRICING (COM BANDAS DE RISCO E INCERTEZA)
# ==============================================================================
cat("\n[2/3] Rodando Simulador de Receita com Regras de Inferência e Risco...\n")
## 
## [2/3] Rodando Simulador de Receita com Regras de Inferência e Risco...
tabela_simulacao <- df_top_cat %>%
  split(.$product_category_name) %>%
  map_df(function(dados_categoria) {
    
    categoria_nome <- unique(dados_categoria$product_category_name)
    
    tryCatch({
      elasticidade_info <- tabela_elasticidade %>% filter(Categoria == categoria_nome)
      if (nrow(elasticidade_info) == 0) return(NULL)
      
      elast_val <- elasticidade_info$Elasticidade
      ic_inf <- elasticidade_info$CI_95_Inferior
      ic_sup <- elasticidade_info$CI_95_Superior
      
      # Usando Mediana ao invés de Média (Proteção contra Outliers)
      preco_atual <- median(dados_categoria$preco_medio, na.rm = TRUE)
      qtd_atual <- median(dados_categoria$quantidade, na.rm = TRUE)
      receita_atual <- preco_atual * qtd_atual
      
      # Regra de Negócio baseada no Intervalo de Confiança 
      if (ic_sup < -1) {
        mudanca_preco_pct <- -0.10
        acao <- "Dar Desconto (-10%)"
      } else if (ic_inf > -1) {
        mudanca_preco_pct <- 0.10
        acao <- "Aumentar Preço (+10%)"
      } else {
        mudanca_preco_pct <- 0.00
        acao <- "Manter Preço (Zona de Incerteza)"
      }
      
      # Matemática da Simulação (Preço Novo)
      preco_simulado <- preco_atual * (1 + mudanca_preco_pct)
      
      # O PULO DO GATO: Simulando a demanda nos 3 cenários (Base, IC Inferior, IC Superior)
      qtd_base   <- qtd_atual * (1 + mudanca_preco_pct)^elast_val
      qtd_bound1 <- qtd_atual * (1 + mudanca_preco_pct)^ic_inf
      qtd_bound2 <- qtd_atual * (1 + mudanca_preco_pct)^ic_sup
      
      # Calculando as Receitas Simuladas
      rec_base   <- preco_simulado * qtd_base
      rec_bound1 <- preco_simulado * qtd_bound1
      rec_bound2 <- preco_simulado * qtd_bound2
      
      # Calculando os Impactos Percentuais
      impacto_base <- ((rec_base - receita_atual) / receita_atual) * 100
      
      # Pega o menor valor para o pior cenário e o maior para o melhor cenário
      impacto_pessimista <- ((min(rec_bound1, rec_bound2) - receita_atual) / receita_atual) * 100
      impacto_otimista   <- ((max(rec_bound1, rec_bound2) - receita_atual) / receita_atual) * 100
      
      tibble(
        Categoria = categoria_nome,
        Elasticidade = round(elast_val, 2),
        Acao_Recomendada = acao,
        Impacto_Pessimista = round(impacto_pessimista, 2),
        Impacto_Base = round(impacto_base, 2),
        Impacto_Otimista = round(impacto_otimista, 2)
      )
      
    }, error = function(e) { NULL })
  }) %>%
  arrange(desc(Impacto_Base))
